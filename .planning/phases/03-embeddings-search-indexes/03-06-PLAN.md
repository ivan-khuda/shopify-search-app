```typescript
// The pattern: step.run with deterministic ID, batchErrors accumulator, throw-only-on-full-failure guard
const { errors: upsertErrors } = await step.run(`upsert-batch-${cursorKey}`, async () => {
  const batchErrors: UpsertError[] = [];
  for (const node of batch.products) {
    try {
      await productRepository.upsertProduct(shop, mapToUpsertInput(node));
    } catch (err) {
      batchErrors.push({ shopifyId: node.id, message: err instanceof Error ? err.message : String(err) });
    }
  }
  if (batch.products.length > 0 && batchErrors.length === batch.products.length) {
    throw new Error(`Full batch failed: ${batchErrors.map((e) => e.message).join(', ')}`);
  }
  return { errors: batchErrors };
});

// And the persist-cursor step at lines 101-113 (currently only pushes upsertErrors — to be extended):
await step.run(`persist-cursor-${cursorKey}`, async () => {
  await prisma.syncRun.update({
    where: { id: syncRunId },
    data: {
      cursor: batch.endCursor,
      processedCount: { increment: batch.products.length - upsertErrors.length },
      errors: { push: upsertErrors.map((e) => JSON.stringify(e)) },
    },
  });
});
```
