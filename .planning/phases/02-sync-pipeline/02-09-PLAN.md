```typescript
import { NextResponse } from 'next/server';
import { shopifyClient } from '@/lib/shopify/client';
import { prisma } from '@/lib/db/client';
import { productRepository } from '@/lib/db/repositories/ProductRepository';

export async function POST(req: Request) {
  // STEP 1: Read raw body BEFORE JSON.parse — webhooks.validate requires the unparsed bytes
  const rawBody = await req.text();

  // STEP 2: Validate HMAC via webhooks.validate (NOT utils.validateHmac — D-10 correction)
  const validation = await shopifyClient.webhooks.validate({ rawBody, rawRequest: req });
  if (!validation.valid) {
    return NextResponse.json({ error: 'invalid_hmac' }, { status: 401 });
  }

  const shop = validation.domain;
  const topic = validation.topic;
  const eventId = validation.webhookId; // also available via req.headers.get('x-shopify-webhook-id')

  // STEP 3: Dedup — insert-then-catch-P2002
  try {
    await prisma.webhookEvent.create({ data: { eventId, shop, topic } });
  } catch (err) {
    if ((err as { code?: string }).code === 'P2002') {
      return NextResponse.json({ ok: true }, { status: 200 });
    }
    throw err;
  }

  // STEP 4: Now safe to parse
  const payload = JSON.parse(rawBody);

  // STEP 5: Dispatch by topic
  if (topic === 'products/create' || topic === 'products/update') {
    // SYN-11 conflict resolution
    const existing = await productRepository.findByShopAndHandle(shop, payload.handle);
    if (existing?.updatedAtShopify && payload.updated_at &&
        new Date(payload.updated_at) < existing.updatedAtShopify) {
      return NextResponse.json({ ok: true, skipped: 'stale' }, { status: 200 });
    }
    const mapped = mapWebhookPayloadToUpsertInput(payload); // local helper
    await productRepository.upsertProduct(shop, mapped);
  } else if (topic === 'products/delete') {
    await productRepository.deleteProduct(shop, BigInt(payload.id));
  } else {
    // Unknown topic — already deduped, so just ack
    return NextResponse.json({ ok: true, ignored: topic }, { status: 200 });
  }

  return NextResponse.json({ ok: true }, { status: 200 });
}
```

Important contract notes:
- `validation` is the return value of `webhooks.validate`. Its shape per RESEARCH.md Q4: `{ valid: boolean, domain: string, topic: string, webhookId: string }`. Use `validation.domain` (not `req.headers.get('x-shopify-shop-domain')`) as the canonical shop — the validator already extracted and verified it.
- `payload` from webhook is the REST API shape, NOT the GraphQL shape. Variant prices come as strings; options are at the product root as `{name, values}`; the product `id` is a numeric (not a GID). The mapping function for webhook payload is DIFFERENT from `mapToUpsertInput` in `services/shopify/ShopifyProductService.ts` (which handles GraphQL nodes). Implement `mapWebhookPayloadToUpsertInput` inline at the bottom of this route file (or extract to a small helper inside `services/shopify/`); do NOT reuse the GraphQL mapper.
- Repository addition: D-17 leaves the choice between `productRepository.findByShopAndHandle` (new method) and inline `prisma.product.findFirst({ where: { shop, handle } })`. Chosen: add `findByShopAndHandle` to the repository — keeps multi-tenant safety in one place; matches the pattern of `findByShopAndId` (lines 165-167 of the current file).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Shopify→webhook | Inbound HTTP POST from Shopify (or any internet client); HMAC header is the only authenticity proof |
| webhook→DB | Validated payload writes through productRepository |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-2-spoof | Spoofing | `POST /api/shopify/webhook` | mitigate | `await shopifyClient.webhooks.validate({ rawBody, rawRequest: req })` validates `X-Shopify-Hmac-Sha256` against `SHOPIFY_API_SECRET`. Raw body is read via `req.text()` BEFORE `JSON.parse` to preserve bytes for HMAC computation. `utils.validateHmac` is forbidden (D-10) — that API is for OAuth/App Proxy query-param HMAC. |
| T-2-replay | Tampering | `prisma.webhookEvent` table | mitigate | `WebhookEvent.eventId` is `@id` (unique). `prisma.webhookEvent.create` is wrapped in try/catch; on `P2002` (Prisma unique-violation code) the handler returns 200 without touching the products table. Eventid is derived from `validation.webhookId` (validator extracts `X-Shopify-Webhook-Id` per Shopify's per-delivery uniqueness guarantee). |
| T-2-stale | Tampering | `productRepository.upsertProduct` | mitigate | After dedup but before upsert, the handler calls `productRepository.findByShopAndHandle(shop, payload.handle)`. If `existing.updatedAtShopify` is non-null AND `new Date(payload.updated_at) < existing.updatedAtShopify`, the handler returns 200 with `skipped: 'stale'` and does NOT call upsertProduct. This blocks the race where a webhook for an older Shopify state arrives after a newer sync write. |
| T-2-tenant | Tampering | productRepository | mitigate | Every write uses `shop = validation.domain` from the verified HMAC. The repository signature `upsertProduct(shop, input)` enforces shop scoping (Phase 1 D-03). No code path bypasses the repository. |
| T-2-leak | Information Disclosure | logs | mitigate | No `console.log` of headers, body, or validation result. Verified via `grep -c "console.log"` = 0 acceptance criterion. |
| T-2-iso | Information Disclosure | cross-shop write | mitigate | `findByShopAndHandle(shop, ...)` scopes the conflict-resolution read by shop. A shop's webhook for a handle that exists in another shop's catalog will not touch the other shop's row (separate composite key). |
</threat_model>
