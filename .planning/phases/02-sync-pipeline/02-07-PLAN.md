```typescript
import { NextResponse } from 'next/server';
import { createHash } from 'node:crypto';
import { withShopifySession } from '@/lib/shopify/auth';
import { prisma } from '@/lib/db/client';
import { inngest } from '@/lib/inngest/client';

export const POST = withShopifySession(async ({ shop, session }) => {
  void session; // not needed here; Inngest function reloads its own offline session

  // 1. Compute 5-min-bucket idempotency key
  const idempotencyKey = createHash('sha256')
    .update(`${shop}|${Math.floor(Date.now() / 300_000)}`)
    .digest('hex');

  // 2. Find-or-create
  const existing = await prisma.syncRun.findFirst({
    where: { shop, idempotencyKey },
  });
  if (existing) {
    return NextResponse.json({ syncRunId: existing.id });
  }

  const run = await prisma.syncRun.create({
    data: { shop, idempotencyKey, state: 'queued', processedCount: 0 },
  });

  // 3. Fire Inngest event (returns after Inngest acks; ~100ms)
  await inngest.send({
    name: 'shopify/product.sync',
    data: { syncRunId: run.id, shop },
  });

  return NextResponse.json({ syncRunId: run.id });
});
```

Per D-05: "If a row exists and state in ('queued', 'running'), return the existing syncRunId. If the row exists but is in a terminal state, still return the existing id." The findFirst above already implements this (it doesn't filter by state — any matching idempotencyKey row wins).

Inngest event payload contract (locked via T-2-leak): `data` contains ONLY `{ syncRunId, shop }`. NEVER include `session`, `accessToken`, or any other secret.
