import { NextRequest, NextResponse } from 'next/server';
import { shopifyClient } from '@/lib/shopify/client';

export async function middleware(request: NextRequest): Promise<NextResponse> {
  const { searchParams } = request.nextUrl;
  let shop = searchParams.get('shop');
  const authHeader = request.headers.get('Authorization');
  if (!shop && authHeader?.startsWith('Bearer ')) {
    const token = authHeader.slice(7);
    try {
      const payload = await shopifyClient.session.decodeSessionToken(token);
      shop = new URL(payload.dest as string).hostname;
    } catch {
      return redirectToAuth(request);
    }
  }
  // (commented-out session check)
  return NextResponse.next();
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser navigation → embedded admin page | Untrusted: any unauthenticated user can navigate directly to `/onboarding?shop=X`. Proxy guard verifies the offline session exists for shop X before allowing through. |
| `?shop=` query parameter | The shop value is taken at face value as a lookup key — but unless an offline session was minted for that exact shop (via the OAuth flow), the lookup fails and the user is redirected to `/api/auth`. An attacker cannot inject an arbitrary `?shop=` to bypass auth. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-1-04 | Elevation of Privilege (unauthenticated page access) | proxy.ts matcher on /onboarding and /chat | mitigate | `export const config = { matcher: ['/onboarding/:path*', '/chat/:path*'] }` + session-lookup-or-redirect guard. Unauthenticated navigation triggers a 307 to `/api/auth`. Verified by 4 unit tests. |
| T-1-06 | Spoofing (forged `?shop=` query) | proxy.ts shop derivation | mitigate | The shop param is only useful as a lookup key into the offline-session storage; without a server-minted session for that shop, the lookup returns `undefined` and the user is redirected. The forged shop cannot grant access. |
| T-1-03 | Information Disclosure (token logging in proxy) | proxy.ts | mitigate | Grep gate confirms zero `console.log` in proxy.ts (D-10). Plan 04 already deleted the originals in middleware.ts before this rewrite. |
</threat_model>
