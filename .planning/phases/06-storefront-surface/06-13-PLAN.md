    ```typescript
    const shop = ...;  // from wrapper closure
    const rl = rateLimit(`shop:${shop}`, 'rest');
    if (!rl.ok) return NextResponse.json({ error: 'rate_limited' }, { status: 429, headers: { 'Retry-After': String(rl.retryAfterSeconds) } });
    const manifestPath = path.join(process.cwd(), 'public', 'storefront-manifest.json');
    if (!existsSync(manifestPath)) return NextResponse.json({ error: 'bundle_not_built' }, { status: 500 });
    const manifest = JSON.parse(readFileSync(manifestPath, 'utf8')) as { bundle: string; version: string };
    const host = process.env.HOST;
    if (!host) return NextResponse.json({ error: 'host_not_configured' }, { status: 500 });
    // manifest.bundle is a leading-slash path like '/storefront-bundle-abc.js'
    const bundle = host.replace(/\/$/, '') + manifest.bundle;
    return NextResponse.json({ bundle, version: manifest.version });
    ```

    Export the `mount` and `toggle` API on `window.smartdiscovery`:
    ```typescript
    function mount(opts: { shop: string; customerId: string | null; accent: string; position: 'bottom_right' | 'bottom_left' }): void {
      const rootEl = document.querySelector('smartdiscovery-app');
      if (!rootEl) return;
      // Remove the body's skeleton class — React drawer takes over visual ownership
      document.body.classList.remove('sd-skeleton-open');
      // Resolve visitor_id from localStorage (StorefrontAdapter also does this, but we need it
      // for the DbBacked stores' constructor opts).
      const STORAGE_KEY = 'smartdiscovery.visitor_id';
      let visitorId = window.localStorage.getItem(STORAGE_KEY);
      if (!visitorId) {
        visitorId = crypto.randomUUID();
        window.localStorage.setItem(STORAGE_KEY, visitorId);
      }
      // Build mount container inside the custom element (separate from the FAB the loader painted)
      let container = rootEl.querySelector<HTMLDivElement>('.sd-drawer-mount');
      if (!container) {
        container = document.createElement('div');
        container.className = 'sd-drawer-mount';
        rootEl.appendChild(container);
      }
      if (!reactRoot) {
        reactRoot = createRoot(container);
      }
      lastProps = opts;
      reactRoot.render(
        <StorefrontDrawer
          shop={opts.shop}
          visitorId={visitorId}
          customerId={opts.customerId}
          accent={opts.accent}
          position={opts.position}
          initialOpen={true}  // user just clicked FAB; auto-open on first mount
        />
      );
    }

    function toggle(): void {
      if (drawerToggleHandle) drawerToggleHandle();
    }

    declare global {
      interface Window {
        smartdiscovery?: { mount: typeof mount; toggle: typeof toggle };
      }
    }

    window.smartdiscovery = { mount, toggle };
    ```

    Important: `toggle()` is invoked on subsequent FAB clicks AFTER the first mount. The simplest implementation is for `StorefrontDrawer` to register its open-toggle setter on `drawerToggleHandle` via a useEffect side-effect — OR the entry re-renders the React tree with toggled `initialOpen` (wasteful). The cleanest path: `StorefrontDrawer` accepts an optional `controlRef?: { current: { toggle: () => void } }` prop and the entry passes a stable ref it can call from `toggle()`. Executor implements whichever pattern best fits — flag in code comments.

    Document in the file header: this entry runs only inside the dynamically-imported main bundle, AFTER the loader.js IIFE has painted the FAB. `window.smartdiscovery.mount` is the loader's hook.
