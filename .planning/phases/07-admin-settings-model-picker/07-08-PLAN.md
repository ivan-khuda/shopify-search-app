       ```
       const providerOrder = useMemo(() => {
         const groups = new Map<string, CatalogModel[]>();
         for (const m of catalog) {
           if (!groups.has(m.provider)) groups.set(m.provider, []);
           groups.get(m.provider)!.push(m);
         }
         // sort providers alphabetically
         const sortedProviders = [...groups.keys()].sort();
         const rows: CatalogModel[] = [];
         for (const p of sortedProviders) {
           const group = groups.get(p)!;
           // float active to top within its provider group
           const active = group.find(m => m.id === activeId);
           const rest = group.filter(m => m.id !== activeId);
           if (active) rows.push(active);
           rows.push(...rest);
         }
         return rows;
       }, [catalog, activeId]);
       ```
    5. Sorted rows computation: `const rows = useMemo(() => { if (!sort) return providerOrder; return [...catalog].sort((a, b) => { const av = a[sort.key]; const bv = b[sort.key]; return sort.direction === 'asc' ?     6. `const dirty = selectedId !== activeId;`
